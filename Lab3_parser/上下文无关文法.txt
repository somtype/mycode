program -> 	block
block	->	{ decls stmts }
decls	-> 	decl decls | nul
decl	->	type id;
type 	->	int | float | char
stmts 	-> 	stmt stmts | nul
stmt	->	id = expr ;
            | if ( bool ) stmt
            | if ( bool ) stmt else stmt
            | while ( bool ) stmt
            | do stmt while ( bool ) ;
            | break ;
            | block
bool    ->  expr bool_res
bool_res->	< expr
            | <= expr
            | > expr
            | >= expr
            | nul
expr 	-> 	term expr_res
expr_res->	+ term expr_res
            | - term expr_res
            | nul
term 	->	factor term_res
term_res->	* factor term_res
            | / factor term_res
            | nul
factor 	-> 	( expr ) | id | num
